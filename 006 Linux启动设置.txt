#grub中设置kernel参数
    init=/bin/bash      #进入bash模式
    init=/sysroot/bin/bash  
    singel              #进入单用户模式
    3                   #进入命令行模式
    loglevel=[n]        #0-7  6:KERN_INFO, 7:KERN_DEBUG

    xen_emul_unplug=never   #不拔除xen模拟设备
    xen_platform_pci.dev_unplug=all
    nomodeset           #debian9

#grub1
    重建引导
    #> grub
    #> device (hd0) /dev/xvda
    #> root(hd0,0) 第一块磁盘第一个分区  /boot分区
    #> setup (hd0)

    grub引导
    #> device (hd0) /dev/xvda
    #> root (hd0,0)
    #> kernel xxx root=xxx
    #> initrd xxx
    #> boot
